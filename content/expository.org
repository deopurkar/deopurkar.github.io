  (setq expos nil)
  (setq rest nil)
  (scan-dir-for-id (org-id-get-create nil)
  		 project-root-directory
  		 (lambda (pom)
  		   (let ((tags (org-get-tags pom)))
  		     (print org-file-tags)
  		     (if (or 
  			     (member "expository" tags)
  			     (member "talk" tags)
  			     (member "paper" tags)
  			     (member "preprint" tags))
  			 (cl-pushnew (pretty-print-backlink pom) expos :test 'equal)
  		       (cl-pushnew (pretty-print-backlink pom) rest :test 'equal)))))
  (concat
   " - "
   (string-join 
    expos
    "\n - "))
#+end_src
